Quick Start

import 'lumina-ui/dist/style.css';

function App() {
  return (
    <ThemeProvider defaultTheme="light">
      <Card>
        <h1>Welcome to Lumina UI</h1>
        <Button variant="gradient">Get Started</Button>
      </Card>
    </ThemeProvider>
  );
}
